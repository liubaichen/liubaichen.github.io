#! 矩阵表示感知机

import numpy

#! AND gate
def AND(x1,x2):
    x = numpy.array([x1,x2])
    w = numpy.array([0.5,0.5])
    b = -0.8
    res = numpy.sum(x*w)+b
    if res<=0:
        return 0
    else:
        return 1
    
def NAND(x1,x2):
    x = numpy.array([x1,x2])
    w = numpy.array([0.5,0.5])
    b = -0.8
    res = numpy.sum(x*w)+b
    if res<=0:
        return 1
    else:
        return 0

def OR(x1,x2):
    x = numpy.array([x1,x2])
    w = numpy.array([0.5,0.5])
    b = -0.4
    res = numpy.sum(x*w)+b
    if res<=0:
        return 0
    else:
        return 1
    
#! 使用多层感知机构建异或门
def XOR(x1,x2):
    #? first layer
    s1 = NAND(x1,x2)
    s2 = OR(x1,x2)
    res = AND(s1,s2)
    if res<=0:
        return 0
    else:
        return 1
